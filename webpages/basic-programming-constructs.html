18. Write a C++ program to convert a string to an integer

Answer 18:

#include <iostream>
#include <string>

using namespace std;


int main(){

    string text;
    int sign = 1;
    int index = 0;
    int number = 0;

    cout << "Enter numeric string: ";
    cin >> text;

    if (text.length() == 0){
        cout << "Invalid input ";
        return 0;
    }

    if (text[0] == '-'){
        sign = -1;
        index = 1;
        if (index == (int)text.length()){
        cout << "Invalid input ";
        return 0;
    }
    }

    for (int i = index; i < (int)text.length(); i++){
        if (text[i] < '0' || text[i] > '9'){
            cout << "Invalid input\n";
            return 0;
        }

        number = number * 10 + (text[i] - '0');
    }

    number = number * sign;

    cout << "Converted int value : " << number ;
    return 0;
}
